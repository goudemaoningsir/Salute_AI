## 一、多项式回归原理

多项式回归是一种特殊的线性回归，它利用多项式特征来拟合数据中的非线性关系。尽管本质上仍属于线性模型，但通过引入自变量的多项式项，使其能够更好地描述非线性趋势。

多项式回归模型的一般形式为：

$$
y = \beta_0 + \beta_1 x + \beta_2 x^2 + \beta_3 x^3 + \ldots + \beta_d x^d + \epsilon
$$

其中：

- $ y $ 是响应变量（目标变量）。
- $ x $ 是自变量。
- $ \beta_0, \beta_1, \ldots, \beta_d $ 是模型的参数。
- $ \epsilon $ 是误差项。

在这个模型中，自变量 $ x $ 的多项式项（如 $ x^2, x^3, \ldots, x^d $）被当作新的特征引入，从而使模型能够捕捉数据中的非线性关系。

## 二、为什么使用多项式回归

**捕捉非线性关系**：在实际问题中，许多现象都表现出非线性特征。多项式回归通过引入自变量的多项式项，能够更好地拟合这些非线性关系。例如，经济学中的供需关系、工程学中的应力-应变关系、生物学中的生长曲线等，往往是非线性的。

**灵活性**：多项式回归通过增加多项式的次数（degree），可以变得非常灵活，能够拟合各种形式的非线性数据。这种灵活性使得多项式回归在处理复杂数据时具有显著优势。

**模型扩展**：多项式回归是线性回归的自然扩展，只需将原始特征扩展为多项式特征即可，无需引入复杂的非线性优化技术。这意味着在多项式回归中，我们仍然可以使用线性回归中常用的估计方法（如最小二乘法）来估计模型参数。

## 三、多项式回归实现

以下是一个使用sklearn实现多项式回归的代码示例。

```python
import os
import torch
from torch import nn
import matplotlib.pyplot as plt

# 解决某些环境中的重复库问题
os.environ["KMP_DUPLICATE_LIB_OK"] = "TRUE"

############ 1. 生成数据 ############
# 参数设置
n_train, n_test = 100, 100  # 训练集和测试集的样本数量
true_w, true_b = [1.2, -3.4, 5.6], 5  # 多项式回归的真实权重和偏置
num_epochs = 100  # 训练的轮数
lr = 0.01  # 学习率

# 生成合成数据
features = torch.randn((n_train + n_test, 1))  # 生成标准正态分布的特征
poly_features = torch.cat([features**i for i in range(1, 5)], 1)  # 构造多项式特征
labels = (
    sum(true_w[i] * poly_features[:, i] for i in range(len(true_w))) + true_b
)  # 生成标签
labels += torch.normal(0, 0.01, size=labels.size())  # 加入高斯噪声


# 数据检查（可选）
# print(features[:2], poly_features[:2], labels[:2], features.size())


############ 2. 定义模型 ############
def create_model(input_dim):
    """创建线性回归模型"""
    net = nn.Linear(input_dim, 1)
    return net


############ 3. 定义损失函数和优化器 ############
loss = nn.MSELoss()


def create_optimizer(net, lr=0.01):
    """创建优化器"""
    optimizer = torch.optim.SGD(net.parameters(), lr=lr)
    return optimizer


############ 4. 训练模型 ############
def train_model(
    train_features, train_labels, net, loss, optimizer, num_epochs=100, batch_size=10
):
    dataset = torch.utils.data.TensorDataset(train_features, train_labels)
    train_iter = torch.utils.data.DataLoader(dataset, batch_size, shuffle=True)

    train_ls = []
    for epoch in range(num_epochs):
        for X, y in train_iter:
            optimizer.zero_grad()
            l = loss(net(X), y.reshape(-1, 1))
            l.backward()
            optimizer.step()
        train_ls.append(loss(net(train_features), train_labels.reshape(-1, 1)).item())
    return train_ls


############ 5. 绘制训练曲线 ############
def plot_loss(train_ls, test_ls):
    """绘制训练和测试损失曲线"""
    plt.figure(figsize=(6, 4))
    plt.xlabel("Epochs")
    plt.ylabel("Loss")
    plt.semilogy(range(1, num_epochs + 1), train_ls, label="Train")
    plt.semilogy(range(1, num_epochs + 1), test_ls, label="Test", linestyle=":")
    plt.legend()
    plt.show()


############ 6. 绘制拟合曲线 ############
def plot_fitted_curve(
    train_features, train_labels, test_features, test_labels, net, degree
):
    """绘制拟合曲线"""
    plt.figure(figsize=(10, 6))
    plt.scatter(
        train_features[:, 0].detach().numpy(),
        train_labels.detach().numpy(),
        color="blue",
        label="Training Data",
    )
    plt.scatter(
        test_features[:, 0].detach().numpy(),
        test_labels.detach().numpy(),
        color="green",
        label="Test Data",
    )

    sorted_train_features, sorted_train_indices = torch.sort(train_features[:, 0])
    plt.plot(
        sorted_train_features.detach().numpy(),
        net(train_features[sorted_train_indices]).detach().numpy(),
        color="red",
        label="Fitted Curve",
    )

    plt.legend()
    plt.xlabel("X")
    plt.ylabel("y")
    plt.title(f"Polynomial Regression (Degree={degree})")
    plt.show()


############ 7. 综合运行 ############
def fit_and_plot(train_features, train_labels, test_features, test_labels, degree):
    net = create_model(train_features.shape[-1])
    optimizer = create_optimizer(net, lr)

    train_ls = train_model(
        train_features, train_labels, net, loss, optimizer, num_epochs
    )
    test_ls = [
        loss(net(test_features), test_labels.reshape(-1, 1)).item()
        for _ in range(num_epochs)
    ]

    plot_loss(train_ls, test_ls)
    plot_fitted_curve(
        train_features, train_labels, test_features, test_labels, net, degree
    )


# 进行不同多项式次数的回归并绘图
fit_and_plot(
    poly_features[:n_train, :3],
    labels[:n_train],
    poly_features[n_train:, :3],
    labels[n_train:],
    degree=3,
)  # 三次多项式回归
fit_and_plot(
    poly_features[:n_train, :],
    labels[:n_train],
    poly_features[n_train:, :],
    labels[n_train:],
    degree=4,
)  # 四次多项式回归
fit_and_plot(
    features[:n_train, :],
    labels[:n_train],
    features[n_train:, :],
    labels[n_train:],
    degree=1,
)  # 一次多项式回归（线性回归）
```
